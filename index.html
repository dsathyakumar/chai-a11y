module.exports.configure = function(newOpts) {
    options = defaultsDeep(newOpts || {}, {
        outDir: 'shots/units',
        tolerance: 5,
        port: 4567,
        templateType: 'marko',
        lassoConfig: {
            plugins: [
                'lasso-less',
                'lasso-autoprefixer',
                'lasso-marko'
            ],
            'url-prefix': '/static',
            outputDir: 'static',
            fingerprintsEnabled: false,
            minify: false,
            bundlingEnabled: false,
            resolveCssUrls: true,
            cacheProfile: 'development'
        }
    });
    fs.mkdirsSync(path.join(process.cwd(), options.outDir));
    configured = true;
};
